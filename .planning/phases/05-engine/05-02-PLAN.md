<pitfalls>
From PITFALLS.md:

**PITFALL-4: 1.2GB Download Without Resume**
- Use HTTP Range requests for resume
- Download to `.part` file, rename atomically after verification
- SHA-256 checksum before marking ready
- Store progress in companion `.meta` JSON

**PITFALL-3: mmap Deadlock on Partial Load**
- Never load model file while download in progress
- Always validate file sizes before mmap
- Use .part pattern, atomic rename after verification

**PITFALL-12: HuggingFace LFS Redirects**
- Always resume from original HuggingFace URL, not cached redirect
- Cache ETag, not redirect URL
- Signed CDN tokens may expire
</pitfalls>

    import okhttp3.Request
    import okhttp3.Response
    import java.io.File
    import java.io.FileOutputStream
    import java.security.MessageDigest
    import kotlinx.coroutines.flow.Flow
    import kotlinx.coroutines.flow.flow
    import javax.inject.Inject
    import javax.inject.Singleton
    
    @Singleton
    class HuggingFaceDownloadService @Inject constructor(
        private val okHttpClient: OkHttpClient
    ) {
        companion object {
            const val HF_REPO = "jinjin06/Qwen3.5-2B-MNN"
            const val HF_BASE_URL = "https://huggingface.co"
            
            // Model files to download
            val MODEL_FILES = listOf(
                "llm.mnn",
                "llm.mnn.weight",
                "llm_config.json",
                "config.json",
                "tokenizer.model"
            )
        }
        
        /**
         * Downloads a file from HuggingFace with HTTP Range resume support.
         * Per DL-02: Supports resume after interruption.
         * Per DL-04: Downloads to .part file, caller renames after verification.
         * 
         * @param relativePath Path within repo (e.g., "llm.mnn.weight")
         * @param partFile Target .part file to write to
         * @param startByte Byte offset to resume from (0 for fresh download)
         * @param expectedSha256 Expected SHA-256 hash for verification (null if unknown)
         * @return Flow emitting DownloadProgress (bytes, percentage, speed)
         */
        fun downloadWithResume(
            relativePath: String,
            partFile: File,
            startByte: Long = 0L,
            expectedSha256: String? = null
        ): Flow<DownloadProgress> = flow {
            // Build URL: https://huggingface.co/jinjin06/Qwen3.5-2B-MNN/resolve/main/{file}
            val url = "$HF_BASE_URL/$HF_REPO/resolve/main/$relativePath"
            
            // Build Range request if resuming
            val requestBuilder = Request.Builder().url(url)
            if (startByte > 0) {
                requestBuilder.header("Range", "bytes=$startByte-")
            }
            requestBuilder.header("User-Agent", "FoodExpiryApp/2.0") // Per PITFALL-12
            
            val response = okHttpClient.newCall(requestBuilder.build()).execute()
            
            if (!response.isSuccessful && response.code != 206) {
                throw DownloadException("HTTP ${response.code}: ${response.message}")
            }
            
            // Get total size from Content-Range header or Content-Length
            val totalBytes = parseContentRangeTotal(response) ?: response.body?.contentLength() ?: -1L
            val downloadedBytes = startByte
            
            // Stream response body to .part file
            response.body?.byteStream()?.use { input ->
                FileOutputStream(partFile, true).use { output -> // append mode for resume
                    val buffer = ByteArray(8192)
                    var bytesRead: Int
                    var lastProgressTime = System.currentTimeMillis()
                    var bytesSinceLastProgress = 0L
                    
                    while (input.read(buffer).also { bytesRead = it } != -1) {
                        output.write(buffer, 0, bytesRead)
                        bytesSinceLastProgress += bytesRead
                        
                        // Emit progress every 500ms or every 1MB
                        val now = System.currentTimeMillis()
                        if (now - lastProgressTime >= 500 || bytesSinceLastProgress >= 1024 * 1024) {
                            val speed = bytesSinceLastProgress * 1000 / (now - lastProgressTime)
                            val percentage = if (totalBytes > 0) (downloadedBytes + bytesSinceLastProgress) * 100 / totalBytes else 0
                            emit(DownloadProgress(
                                downloadedBytes = downloadedBytes + bytesSinceLastProgress,
                                totalBytes = totalBytes,
                                percentage = percentage.toInt(),
                                speedBytesPerSec = speed
                            ))
                            lastProgressTime = now
                            bytesSinceLastProgress = 0L
                        }
                    }
                }
            }
            
            // Final progress emission
            emit(DownloadProgress(
                downloadedBytes = downloadedBytes + partFile.length(),
                totalBytes = totalBytes,
                percentage = 100,
                speedBytesPerSec = 0
            ))
        }
        
        /**
         * Verifies SHA-256 checksum of downloaded file.
         * Per DL-03: Model files verified before use.
         */
        fun verifySha256(file: File, expectedHash: String): Boolean {
            val digest = MessageDigest.getInstance("SHA-256")
            file.inputStream().use { input ->
                val buffer = ByteArray(8192)
                var bytesRead: Int
                while (input.read(buffer).also { bytesRead = it } != -1) {
                    digest.update(buffer, 0, bytesRead)
                }
            }
            val actualHash = digest.digest().joinToString("") { "%02x".format(it) }
            return actualHash == expectedHash
        }
        
        private fun parseContentRangeTotal(response: Response): Long? {
            val contentRange = response.header("Content-Range")
            if (contentRange == null) return null
            // Format: "bytes 0-1234567/1234568"
            val parts = contentRange.split("/")
            return parts.lastOrNull()?.toLongOrNull()
        }
    }
    
    data class DownloadProgress(
        val downloadedBytes: Long,
        val totalBytes: Long,
        val percentage: Int,
        val speedBytesPerSec: Long
    ) {
        val etaSeconds: Long = if (speedBytesPerSec > 0 && totalBytes > downloadedBytes) {
            (totalBytes - downloadedBytes) / speedBytesPerSec
        } else 0L
        
        val downloadedMB: Double = downloadedBytes / (1024.0 * 1024.0)
        val totalMB: Double = totalBytes / (1024.0 * 1024.0)
        val speedMBPerSec: Double = speedBytesPerSec / (1024.0 * 1024.0)
    }
    
    class DownloadException(message: String) : Exception(message)
    ```

    - SHA-256 verification method (per DL-03)

    package com.example.foodexpiryapp.data.local
    
    import android.content.Context
    import java.io.File
    import javax.inject.Inject
    import javax.inject.Singleton
    
    @Singleton
    class ModelStorageManager @Inject constructor(
        private val context: Context
    ) {
        companion object {
            const val MODEL_DIR_NAME = "mnn_models"
            const val PART_EXTENSION = ".part"
            const val META_EXTENSION = ".meta"
        }
        
        /**
         * Gets the model storage directory.
         * Per PITFALL-4: Use external files dir to avoid internal storage quota.
         */
        fun getModelDirectory(): File {
            val dir = File(context.getExternalFilesDir(null), MODEL_DIR_NAME)
            if (!dir.exists()) dir.mkdirs()
            return dir
        }
        
        /**
         * Gets path for .part file during download.
         * Per DL-04: Partial downloads use .part file pattern.
         */
        fun getPartFilePath(relativePath: String): File {
            return File(getModelDirectory(), "$relativePath$PART_EXTENSION")
        }
        
        /**
         * Gets path for final model file after verification.
         */
        fun getFinalFilePath(relativePath: String): File {
            return File(getModelDirectory(), relativePath)
        }
        
        /**
         * Atomically renames .part to final file after SHA-256 verification.
         * Per DL-04: Atomic rename after verification.
         * Per PITFALL-3: Prevents mmap deadlock on partial files.
         */
        fun finalizePartFile(relativePath: String): Boolean {
            val partFile = getPartFilePath(relativePath)
            val finalFile = getFinalFilePath(relativePath)
            
            if (!partFile.exists()) return false
            
            // Delete existing final file if present (re-download case)
            if (finalFile.exists()) finalFile.delete()
            
            return partFile.renameTo(finalFile)
        }
        
        /**
         * Checks if all required model files exist and are complete.
         */
        fun areAllModelFilesReady(): Boolean {
            val requiredFiles = listOf(
                "llm.mnn",
                "llm.mnn.weight",
                "llm_config.json",
                "config.json",
                "tokenizer.model"
            )
            
            return requiredFiles.all { relativePath ->
                getFinalFilePath(relativePath).exists() && 
                !getPartFilePath(relativePath).exists() // No incomplete downloads
            }
        }
        
        /**
         * Gets total size of downloaded model files.
         */
        fun getTotalDownloadedSize(): Long {
            return getModelDirectory().listFiles()
                ?.filter { !it.name.endsWith(PART_EXTENSION) }
                ?.sumOf { it.length() }
                ?: 0L
        }
        
        /**
         * Cleans up incomplete .part files from interrupted downloads.
         * Per PITFALL-3: Delete incomplete files on startup.
         */
        fun cleanupIncompleteDownloads() {
            getModelDirectory().listFiles()
                ?.filter { it.name.endsWith(PART_EXTENSION) }
                ?.forEach { it.delete() }
        }
        
        /**
         * Deletes all model files (for re-download).
         */
        fun deleteAllModelFiles() {
            getModelDirectory().listFiles()?.forEach { it.delete() }
        }
    }
    ```

    - cleanupIncompleteDownloads() for startup safety (per PITFALL-3)
